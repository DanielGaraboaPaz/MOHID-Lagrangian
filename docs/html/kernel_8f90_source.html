319  function beaching(self, sv, svDt)
320  class(kernel_class), intent(inout) :: self
321  type(statevector_class), intent(inout) :: sv
322  real(prec), dimension(size(sv%state,1),size(sv%state,2)), intent(in) :: svdt
323  real(prec), dimension(size(sv%state,1),size(sv%state,2)) :: beaching
324  real(prec), dimension(size(sv%state,1)) :: beachcoeff
325  real(prec), dimension(size(sv%state,1)) :: beachcoeffrand
326  real(prec), dimension(size(sv%state,1)) :: beachweight
327  real(prec) :: lbound, ubound
328  integer :: i
329 
330  beachcoeff = 1.0
331  call random_number(beachcoeffrand) !this is a uniform distribution generator
332  beachcoeffrand = max(0.0, beachcoeffrand - globals%Constants%BeachingStopProb) !clipping the last % to zero
333  beachcoeffrand = beachcoeffrand*(1.0/max(maxval(beachcoeffrand),1.0)) !normalizing
334 
335  beaching = svdt
336 
337  !getting the bounds for the interpolation of the land interaction field that correspond to beaching
338  lbound = (globals%Mask%beachVal + globals%Mask%waterVal)*0.5
339  ubound = (globals%Mask%beachVal + globals%Mask%landVal)*0.5
340 
341  !beachWeight = 1 - 0.5*(sv%landIntMask - lbound)/(ubound-lbound) !linear distance weight for beaching
342  beachweight = 1 - 0.9*(sv%landIntMask - lbound)/(ubound-lbound)*(sv%landIntMask - lbound)/(ubound-lbound) !quadratic weight
343 
344  !replacing 1.0 with a coefficient from beaching where needed
345  where(sv%landIntMask <= ubound .and. sv%landIntMask >= lbound) beachcoeff = beachcoeffrand*beachweight
346  do i=1,3
347  beaching(:,i) = svdt(:,i)*beachcoeff !position derivative is affected
348  sv%state(:,i+3) = sv%state(:,i+3)*beachcoeff !so are the velocities
349  end do

351  end function beaching
