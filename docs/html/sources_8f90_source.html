421  sposi(i,2) = (1.0-weight)*posi(index,2) + weight*posi(index+1,2)
422  sposi(i,3) = (1.0-weight)*posi(index,3) + weight*posi(index+1,3)
423  end if
424  end do
425  end if
426  end do
427  allocate(self%par%variable_position(size(stime)))
428  do i=1, size(stime)

430  end do
