294

297  call csvreader%getColumn(ratefile, filename, 2, rate)
298  call csvreader%closeFile(ratefile)
299  !interpolating the csv data to a regular dt spaced array and storing the data in the Source
300  allocate(stime(int(min(globals%Parameters%TimeMax,maxval(time))/globals%SimDefs%dt)+1))

304  end do

307  srate(i) = 0.0
308  else
309  do index=1, size(time)-1
310  if (stime(i)>=time(index)) then
311  weight = (stime(i)-time(index))/(time(index+1)-time(index));
312  srate(i) = (1.0-weight)*rate(index) + weight*rate(index+1);
313  end if
314  end do
315  end if
