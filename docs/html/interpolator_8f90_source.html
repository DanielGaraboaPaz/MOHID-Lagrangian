140  function interp4d(self, x, y, z, t, out, field, n_fv, n_cv, n_pv, n_tv, n_e)
141  class(interpolator_class), intent(in) :: self
142  real(prec), dimension(n_e),intent(in):: x, y, z
143  real(prec), intent(in) :: t
144  logical, dimension(:), intent(in) :: out
145  real(prec), dimension(n_fv, n_cv, n_pv, n_tv), intent(in) :: field
146  integer, intent(in) :: n_fv, n_cv, n_pv, n_tv
147  integer, intent(in) :: n_e
148  integer, dimension(n_e) :: x0, y0, z0, x1, y1, z1
149  real(prec), dimension(n_e) :: xd, yd, zd, c000, c100, c010, c110, c001
150  real(prec), dimension(n_e) :: c101, c011, c111, c00, c10, c01, c11, c0, c1
151  real(prec) :: td
152  integer :: i, t0, t1
153  real(prec), dimension(n_e) :: interp4d
154 
155  ! From x,y,z,t in array coordinates, find the the box inside the field where the particle is
156  x0 = floor(x)
157  y0 = floor(y)
158  z0 = floor(z)
159  t0 = floor(t)
160  x1 = ceiling(x)
161  y1 = ceiling(y)
162  z1 = ceiling(z)
163  t1 = ceiling(t)
164 
165  ! Compute the "normalized coordinates" of the particle inside the data field box
166  xd = (x-x0)/(x1-x0)
167  yd = (y-y0)/(y1-y0)
168  zd = (z-z0)/(z1-z0)
169  td = (t-t0)/(t1-t0)
170 
171  where (out)
172  x0 = 1.0
173  x1 = 1.0
174  y0 = 1.0
175  y1 = 1.0
176  z0 = 1.0
177  z1 = 1.0
178  end where
179 
180  ! In case that particle is on a point box, we set it to 0 to avoid inf errors
181  where (x1 == x0) xd = 0.
182  where (y1 == y0) yd = 0.
183  where (z1 == z0) zd = 0.
184  if (t1 == t0) td = 0.
185 
186  ! Interpolation on the first dimension and collapse it to a three dimension problem
187  forall(i=1:n_e)
188  c000(i) = field(x0(i),y0(i),z0(i),t0)*(1.-xd(i)) + field(x1(i),y0(i),z0(i),t0)*xd(i) !y0x0z0t0! y0x1z0t0
189  c100(i) = field(x0(i),y1(i),z0(i),t0)*(1.-xd(i)) + field(x1(i),y1(i),z0(i),t0)*xd(i)
190  c010(i) = field(x0(i),y0(i),z1(i),t0)*(1.-xd(i)) + field(x1(i),y0(i),z1(i),t0)*xd(i)
191  c110(i) = field(x0(i),y1(i),z1(i),t0)*(1.-xd(i)) + field(x1(i),y1(i),z1(i),t0)*xd(i)
192 
193  c001(i) = field(x0(i),y0(i),z0(i),t1)*(1.-xd(i)) + field(x1(i),y0(i),z0(i),t1)*xd(i) !y0x0z0t0! y0x1z0t0
194  c101(i) = field(x0(i),y1(i),z0(i),t1)*(1.-xd(i)) + field(x1(i),y1(i),z0(i),t1)*xd(i)
195  c011(i) = field(x0(i),y0(i),z1(i),t1)*(1.-xd(i)) + field(x1(i),y0(i),z1(i),t1)*xd(i)
196  c111(i) = field(x0(i),y1(i),z1(i),t1)*(1.-xd(i)) + field(x1(i),y1(i),z1(i),t1)*xd(i)
197  end forall
198  ! Interpolation on the second dimension and collapse it to a two dimension problem
199  c00 = c000*(1.-yd)+c100*yd
200  c10 = c010*(1.-yd)+c110*yd
201  c01 = c001*(1.-yd)+c101*yd
202  c11 = c011*(1.-yd)+c111*yd
203  ! Interpolation on the third dimension and collapse it to a one dimension problem
204  c0 = c00*(1.-zd)+c10*zd
205  c1 = c01*(1.-zd)+c11*zd
206  ! Interpolation on the time dimension and get the final result.
207  interp4d = c0*(1.-td)+c1*td
208  where (out) interp4d = 0.0
209 
210  end function interp4d
