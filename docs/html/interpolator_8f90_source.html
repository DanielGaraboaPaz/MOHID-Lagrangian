119  function interp4d(self, x, y, z, t, field, n_fv, n_cv, n_pv, n_tv, n_e)
120  class(interpolator_class), intent(in) :: self
121  real(prec), dimension(n_e),intent(in):: x, y, z
122  real(prec), intent(in) :: t
123  real(prec), dimension(n_fv, n_cv, n_pv, n_tv), intent(in) :: field
124  integer, intent(in) :: n_fv, n_cv, n_pv, n_tv
125  integer, intent(in) :: n_e
126  integer, dimension(n_e) :: x0, y0, z0, x1, y1, z1
127  real(prec), dimension(n_e) :: xd, yd, zd, c000, c100, c010, c110, c001
128  real(prec), dimension(n_e) :: c101, c011, c111, c00, c10, c01, c11, c0, c1
129  real(prec) :: td
130  integer :: i, t0, t1
131  real(prec), dimension(n_e) :: interp4d
132 
133  ! From x,y,z,t in array coordinates, find the the box inside the field where the particle is
134  x0 = floor(x)
135  y0 = floor(y)
136  z0 = floor(z)
137  t0 = floor(t)
138  x1 = ceiling(x)
139  y1 = ceiling(y)
140  z1 = ceiling(z)
141  t1 = ceiling(t)
142 
143  ! Compute the "normalized coordinates" of the particle inside the data field box
144  xd = (x-x0)/(x1-x0)
145  yd = (y-y0)/(y1-y0)
146  zd = (z-z0)/(z1-z0)
147  td = (t-t0)/(t1-t0)
148 
149  ! In case that particle is on a point box, we set it to 0 to avoid inf errors
150  where (x1 == x0) xd = 0.
151  where (y1 == y0) yd = 0.
152  where (z1 == z0) zd = 0.
153  if (t1 == t0) td = 0.
154 
155  ! Interpolation on the first dimension and collapse it to a three dimension problem
156  forall(i=1:n_e)
157  c000(i) = field(x0(i),y0(i),z0(i),t0)*(1.-xd(i)) + field(x1(i),y0(i),z0(i),t0)*xd(i) !y0x0z0t0! y0x1z0t0
158  c100(i) = field(x0(i),y1(i),z0(i),t0)*(1.-xd(i)) + field(x1(i),y1(i),z0(i),t0)*xd(i)
159  c010(i) = field(x0(i),y0(i),z1(i),t0)*(1.-xd(i)) + field(x1(i),y0(i),z1(i),t0)*xd(i)
160  c110(i) = field(x0(i),y1(i),z1(i),t0)*(1.-xd(i)) + field(x1(i),y1(i),z1(i),t0)*xd(i)
161 
162  c001(i) = field(x0(i),y0(i),z0(i),t1)*(1.-xd(i)) + field(x1(i),y0(i),z0(i),t1)*xd(i) !y0x0z0t0! y0x1z0t0
163  c101(i) = field(x0(i),y1(i),z0(i),t1)*(1.-xd(i)) + field(x1(i),y1(i),z0(i),t1)*xd(i)
164  c011(i) = field(x0(i),y0(i),z1(i),t1)*(1.-xd(i)) + field(x1(i),y0(i),z1(i),t1)*xd(i)
165  c111(i) = field(x0(i),y1(i),z1(i),t1)*(1.-xd(i)) + field(x1(i),y1(i),z1(i),t1)*xd(i)
166  end forall
167  ! Interpolation on the second dimension and collapse it to a two dimension problem
168  c00 = c000*(1.-yd)+c100*yd
169  c10 = c010*(1.-yd)+c110*yd
170  c01 = c001*(1.-yd)+c101*yd
171  c11 = c011*(1.-yd)+c111*yd
172  ! Interpolation on the third dimension and collapse it to a one dimension problem
173  c0 = c00*(1.-zd)+c10*zd
174  c1 = c01*(1.-zd)+c11*zd
175  ! Interpolation on the time dimension and get the final result.
176  interp4d = c0*(1.-td)+c1*td
177  end function interp4d

interpolator_mod::interp4d
real(prec) function, dimension(n_e) interp4d(self, x, y, z, t, field, n_fv, n_cv, n_pv, n_tv, n_e)
method to interpolate a particle position in a given data box based on array coordinates. 4d interpolation is a weighted average of 16 neighbors. Consider the 4D domain between the 16 neighbors. The hypercube is divided into 16 sub-hypercubes by the point in question. The weight of each neighbor is given by the volume of the opposite sub-hypercube, as a fraction of the whole hypercube.
