352  subroutine check(self)
353  implicit none
354  class(parameters_t), intent(inout) :: self
355  type(string) :: outext
356  type(datetime) :: temp
357  type(timedelta) :: simtime
358 
359  if ( any(self%IntegratorIndexes == self%Integrator)) then
360  else
361  outext = '[Globals::parameters::check]: Integrator not recognized, stoping'
362  call log%put(outext)
363  stop
364  end if
365  if ( any(self%OutputFormatIndexes == self%OutputFormat)) then
366  if (self%OutputFormat == 1) then
367  outext = '[Globals::parameters::check]: NetCDF is not implemented yet, try something nicer like VTK, stoping'
368  call log%put(outext)
369  stop
370  end if
371  else
372  outext = '[Globals::parameters::check]: OutputFormat not recognized, stoping'
373  call log%put(outext)
374  stop
375  end if
376  temp = datetime() !default initialization
377  !add new parameters to this search
378  if (self%TimeOut==mv) then
379  outext = '[Globals::parameters::check]: sampling rate parameter (TimeOut) is not set, stoping'
380  call log%put(outext)
381  stop
382  elseif (self%StartTime==temp) then
383  outext = '[Globals::parameters::check]: start time parameter (StartTime) is not set or invalid, stoping'
384  call log%put(outext)
385  stop
386  elseif (self%EndTime==temp) then

388  call log%put(outext)
389  stop
390  end if
391  !Build timemax from the difference between start and end time
392  simtime = self%EndTime - self%StartTime
