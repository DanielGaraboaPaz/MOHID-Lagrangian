70  subroutine loaddatafromstack(self, bBox, blocks)

72  type(boundingbox_class), intent(in) :: bBox
73  type(block_class), dimension(:), intent(inout) :: blocks
74  type(background_class) :: tempBkgd
75  integer :: i, j
76  integer :: fNumber
77  real(prec) :: tempTime(2)
78  logical :: needToRead, appended
79 
80  needtoread = .false.
81  if (self%useInputFiles) then
82  !check if we need to import data (current time and buffer size)
83  if (self%lastReadTime <= globals%SimTime%CurrTime + self%BufferSize/4.0) needtoread = .true.
84  if (self%lastReadTime >= globals%SimTime%TimeMax) needtoread = .false.
85  if (needtoread) then
86  call self%resetReadStatus()
87  !check what files on the stack are to read to backgrounds
88  do i=1, size(self%currentsInputFile)
89  if (self%currentsInputFile(i)%endTime >= globals%SimTime%CurrTime) then
90  if (self%currentsInputFile(i)%startTime <= globals%SimTime%CurrTime + self%BufferSize) then
91  if (.not.self%currentsInputFile(i)%used) self%currentsInputFile(i)%toRead = .true.
92  end if
93  end if
94  end do
95  !read selected files
96  do i=1, size(self%currentsInputFile)
97  if (self%currentsInputFile(i)%toRead) then
98  !import data to temporary background
99  tempbkgd = self%getFullFile(self%currentsInputFile(i)%name)
100  self%currentsInputFile(i)%used = .true.
101  do j=1, size(blocks)
102  if (.not.allocated(blocks(j)%Background)) allocate(blocks(j)%Background(1))
103  !slice data by block and either join to existing background or add a new one
104  if (blocks(j)%Background(1)%initialized) call blocks(j)%Background(1)%append(tempbkgd%getHyperSlab(blocks(j)%extents), appended)
105  if (.not.blocks(j)%Background(1)%initialized) blocks(j)%Background(1) = tempbkgd%getHyperSlab(blocks(j)%extents)
106 
107  !save last time already loaded
108  temptime = blocks(j)%Background(1)%getDimExtents(globals%Var%time)
109  self%lastReadTime = temptime(2)
110 
111  end do
112  !clean out the temporary background data (this structure, even tough it is a local variable, has pointers inside)
113  call tempbkgd%finalize()
114  end if
115  end do
116  end if
117  end if
118 
119  end subroutine loaddatafromstack
