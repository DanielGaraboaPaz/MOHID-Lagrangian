82  subroutine maketaylorgreen(self, res, testbox, testbackground)
83  class(testmaker_class), intent(inout) :: self
84  integer, intent(in) :: res
85  type(box), intent(in) :: testbox

88  type(generic_field_class) :: gfield1, gfield2, gfield3
89  type(string) :: name, units
90  real(prec), allocatable, dimension(:) :: sx, sy, sz, t
91  real(prec), allocatable, dimension(:,:,:,:) :: vx, vy, vz
92  integer :: i, j, m
93  real(prec) :: pi = 4*atan(1.0)
94  real(prec) :: sst
95 
96  allocate(sx(res), sy(res), sz(res), t(res))
97  allocate(vx(res,res,res,res))
98  allocate(vy(res,res,res,res))
99  allocate(vz(res,res,res,res))
100  !create scaled coordinate arrays
101  do i=1, res
102  sx(i) = (i-1)*2.0*pi/(res-1)
103  sz(i) = i-1
104  end do
105  sy = sx
106  t = sz
107  !create the velocity fields in time and space
108  do m=1, res
109  do i=1, res
110  do j=1, res
111  sst = 1.0 - m/res
112  vx(i,j,:,m) = cos(sx(i))*sin(sy(j))*sst
113  vy(i,j,:,m) = -sin(sx(i))*cos(sy(j))*sst
114  end do
115  end do
116  end do
117  vz = 0.0
118  !put the data on generic fields
119  units = 'm/s'
120  call gfield1%initialize(globals%Var%u, units, vx)
121  call gfield2%initialize(globals%Var%v, units, vy)
122  call gfield3%initialize(globals%Var%w, units, vz)
123  !create the dimensions for the background
124  allocate(testbackgroundims(4))
125  units = 'deg'
126  sx = sx*testbox%size%x/(2.0*pi) + testbox%pt%x
127  call testbackgroundims(1)%initialize(globals%Var%lon, units, 1, sx)
128  units = 'deg'
129  sy = sy*testbox%size%y/(2.0*pi) + testbox%pt%y
130  call testbackgroundims(2)%initialize(globals%Var%lat, units, 1, sy)
131  units = 'm'
132  sz = sz*testbox%size%z/(res-1) + testbox%pt%z
133  call testbackgroundims(3)%initialize(globals%Var%level, units, 1, sz)
134  units = 'seg'
135  t = t*globals%Parameters%TimeMax/(res-1)
136  call testbackgroundims(4)%initialize(globals%Var%time, units, 1, t)
137  !construct background
138  name = 'Taylor-Green test'
139  testbackground = background(1, name, testbox, testbackgroundims)
140  call testbackground%add(gfield1)
141  call testbackground%add(gfield2)
142  call testbackground%add(gfield3)
