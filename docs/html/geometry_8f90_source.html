357  function sphere_np_count(dp, r) result(np)
358  implicit none
359  real(prec), intent(in) :: dp
360  real(prec), intent(in) :: r
361  integer :: np
362  integer :: i, j, k, n
363  type(vector) :: pts
364  np=0
365  n=int(3*r/dp)
366  do i=1, n
367  do j=1, n
368  do k=1, n
369  pts = dp*(ex*(i-1)+ey*(j-1)+ez*(k-1)) - r*(ex+ey+ez)
370  if (pts%normL2() .le. r) then
371  np=np+1
372  end if
373  end do
374  end do
375  end do

378  end if
