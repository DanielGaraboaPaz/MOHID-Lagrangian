534  function sphere_np_count(dp, r) result(np)
535  type(vector), intent(in) :: dp
536  real(prec), intent(in) :: r
537  integer :: np
538  integer :: i, j, k, nx, ny, nz
539  type(vector) :: pts
540  np=0
541  nx=int(3*r/dp%x)
542  ny=int(3*r/dp%y)
543  nz=int(3*r/dp%z)
544  do i=1, nx
545  do j=1, ny
546  do k=1, nz
547  pts = (ex*(i-1)*dp%x +ey*(j-1)*dp%y +ez*(k-1)*dp%z) - r*(ex+ey+ez)
548  if (pts%normL2() <= r) then
549  np=np+1
550  end if
551  end do
552  end do
553  end do
554  if (np == 0) then !Just the center point
555  np=1
556  end if
557  end function sphere_np_count
