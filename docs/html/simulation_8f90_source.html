82  implicit none
83  class(simulation_class), intent(inout) :: self
84  type(string) :: outext, aux
85 
86  !integer :: testmat(10,12)
87 
88  outext = '====================================================================='
89  call log%put(outext,.false.)
90  outext = '->Simulation starting'
91  call log%put(outext)
92  outext = '====================================================================='
93  call log%put(outext,.false.)
94 
95  !call writeTestmatrix(testmat)
96 
97  !main time cycle
98  do while (globals%SimTime%CurrTime .lt. globals%Parameters%TimeMax)
99  call globals%Sim%increment_numdt()
100  call self%timerTotalRun%Tic()
101  !activate suitable Sources
102  call self%ToggleSources()
103  !emitt Tracers from active Sources
104  call self%BlocksEmitt()
105  !Distribute Tracers and Sources by Blocks
106  call self%BlocksDistribute()
107  !Optimize Block Tracer lists
108  call self%BlocksConsolidateArrays()
109  !Build AoT
110  call self%BlocksTracersToAoT()
111  !load hydrodynamic fields from files (curents, wind, waves, ...)
112 
113  !Update all tracers with base behavior (AoT) - Integration step
114  if (globals%Sim%getnumdt() /= 1 ) call self%BlocksRunSolver()
115  !AoT to Tracers
116  call self%BlocksAoTtoTracers()
117  !Update Tracers with type-specific behavior
118 
119  !Write results if time to do so
120  call self%OutputStepData()
121  !Print some stats from the time step
122  call self%printTracerTotals()
123  !Clean AoT
124  call self%BlocksCleanAoT()
125  !update Simulation time
126  if (globals%Sim%getnumdt() /= 1 ) call globals%SimTime%setCurrDateTime(globals%SimDefs%dt)
127  !print*, 'Global time is ', Globals%SimTime%CurrTime
128  !print*, 'Can we continue?'
129  !read (*,*)
130  call self%timerTotalRun%Toc()
131  enddo
132  call self%setTracerMemory()
133  call simmemory%detailedprint()
134 
135  call self%timerTotalRun%print()
136  call self%timerPrep%print()
137  call self%timerAoTOps%print()
138  call self%timerSolver%print()
139  call self%timerOutput%print()

160 
161  aux = 'Simulation::Total'

163  aux = 'Simulation::Output'
164  call self%timerOutput%initialize(aux)
165  aux = 'Simulation::Preparation'
166  call self%timerPrep%initialize(aux)
167  aux = 'Simulation::AoT encoding/decoding'
168  call self%timerAoTOps%initialize(aux)
169  aux = 'Simulation::Solver'
170  call self%timerSolver%initialize(aux)
171 
172  ! Initialize logger
173  call log%initialize(outpath)
174  !Print licences and build info
175  call printlicpreamble
176  !initialize parallel services manager
177  call ompmanager%initialize()
178  !initializing memory log
179  call simmemory%initialize()
180  !setting every global variable and input parameter to their default
181  call globals%initialize(outpath = outpath)
182  !initializing geometry class
183  call geometry%initialize()
184  !Check if case file has .xml extension
185  if (casefilename%extension() == '.xml') then
186  ! Initialization routines to build the simulation from the input case file
187  call initfromxml(casefilename)
188  else
189  outext='[initSimulation]: only .xml input files are supported at the time. Stopping'
190  call log%put(outext)
191  stop
192  endif
193  !Case was read and now we can build/initialize our simulation objects that are case-dependent
194  !initilize simulation bounding box
195  call bbox%initialize()
196  !decomposing the domain and initializing the Simulation Blocks
197  call self%decompose()
198  !Distributing Sources
199  call self%setInitialState()
200  !printing memory occupation at the time
201  call simmemory%detailedprint()
202  !Initializing output file streamer
203  call self%OutputStreamer%initialize()
204  !Writing the domain to file
205  call self%OutputStreamer%WriteDomain(globals%Names%casename, bbox, geometry%getnumPoints(bbox), dblock)
206 
207  call self%timerInit%Toc()
208  call self%timerInit%print()
209 
210  end subroutine initsimulation
