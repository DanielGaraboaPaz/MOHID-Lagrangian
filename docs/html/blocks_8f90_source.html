391  subroutine setblocks(auto, nblk, nxi, nyi)
392  implicit none
393  logical, intent(in) :: auto
394  integer, intent(in) :: nblk
395  integer, intent(out) :: nxi, nyi

400 
401  if (auto) then
402  ar = bbox%size%x/bbox%size%y
403  ar = get_closest_twopow(ar) !aspect ratio of our bounding box
404  nyi = sqrt(nblk/ar)
405  if (nyi == 0) then
406  temp(1) = ar
407  outext='[setBlocks]: block auto sizing failed. Bouding box aspect ratio = '//temp(1)//'. Stoping'
408  call log%put(outext)
409  stop
410  endif
411  nxi = (nblk/nyi)
412 
413  b=1
414  do i=1, nxi
415  do j=1, nyi
416  tempbox%pt = bbox%pt + bbox%size%x*(i-1)/nxi*ex + bbox%size%y*(j-1)/nyi*ey - bbox%pt%z*ez
417  tempbox%size = bbox%size%x/nxi*ex + bbox%size%y/nyi*ey
418  call dblock(b)%initialize(b, tempbox)
419  b=b+1
420  end do
421  end do
422  temp(1) = nxi
423  temp(2) = nyi
424  outext='-->Automatic domain decomposition sucessful. Domain is '//temp(1)// ' X ' //temp(2)//' Blocks'
425  call log%put(outext,.false.)
426  end if
427  globals%SimDefs%blocksize = dblock(1)%extents%size
